
### git 小脚本
```linux
#!/bin/bash

echo "+----------------------------------------+"
echo "|    创建当前server的tag并推送至origin   |"
echo "+----------------------------------------+"

# 获取当前目录名
current_dir=$(basename "$(pwd)")

# 获取当前开发分支
current_branch=$(git rev-parse --abbrev-ref HEAD | sed "s/\//-/g")

# 生成 Tag 名称
tag_name="test-${current_dir}-${current_branch}-$(date +%Y%m%d%H%M%S)"

echo $tag_name

# 创建 Tag
git tag "$tag_name"

# 推送 Tag 到远端
git push origin "$tag_name"

echo "------成功------"
```
